### 5.5 Pathfinding (A* Algorithm)

```typescript
function findPath(
  start: Position,
  goal: Position,
  grid: GridCell[][]
): Position[] {
  // A* implementation
  const openSet = new PriorityQueue<Node>();
  const closedSet = new Set<string>();
  
  openSet.enqueue({ pos: start, g: 0, h: heuristic(start, goal), parent: null });
  
  while (!openSet.isEmpty()) {
    const current = openSet.dequeue();
    
    if (current.pos.x === goal.x && current.pos.y === goal.y) {
      return reconstructPath(current);
    }
    
    closedSet.add(posKey(current.pos));
    
    for (const neighbor of getNeighbors(current.pos, grid)) {
      if (closedSet.has(posKey(neighbor))) continue;
      if (!isWalkable(neighbor, grid)) continue;
      
      const g = current.g + 1;
      const h = heuristic(neighbor, goal);
      
      openSet.enqueue({ pos: neighbor, g, h, parent: current });
    }
  }
  
  return []; // No path found
}

function heuristic(a: Position, b: Position): number {
  return Math.abs(a.x - b.x) + Math.abs(a.y - b.y); // Manhattan distance
}
```
